## 002-OOP

`创建对象的方式`
--

```js
var obj = {
    name: 'Jack',
    age: 18,
    Hi: function () { console.log('hello world') }
}
```

```js
var obj = new Object()
obj.name = 'Jack'
obj.age = 18
obj.Hi = function () { console.log('hello world') }
```

```js
function createObj(name, age) {
    var obj = {}
    obj.name = name
    obj.age = age
    obj.Hi = function () { console.log('hello world') }
    return obj
}
```

```js
function createObj(name, age) {
    this.name = name
    this.age = age
    this.Hi = function () { console.log('hello world') }
}
```
